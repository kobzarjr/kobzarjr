//  ПАРСИНГ ВОПРОСОВ (разбиваем по последнему " или ")
// ============================================================
function parseQuestion(raw) {
    const separator = " или ";
    const lastIndex = raw.lastIndexOf(separator);
    if (lastIndex === -1) {
        return { question: raw, optionA: raw, optionB: "?" };
    }
    const a = raw.substring(0, lastIndex).trim();
    let b = raw.substring(lastIndex + separator.length).trim();
    if (b.endsWith('?')) b = b.slice(0, -1).trim();
    if (!b) b = "???";
    return { question: raw, optionA: a, optionB: b };
}

function prepareCategoryData(categoryName) {
    const rawQuestions = rawData[categoryName];
    if (!rawQuestions) return [];
    return rawQuestions.map(raw => parseQuestion(raw));
}

const categoryNames = Object.keys(rawData);
const allQuestions = [];
categoryNames.forEach(cat => {
    const items = prepareCategoryData(cat);
    allQuestions.push(...items);
});

let currentCategory = '';
let currentQuestions = [];
let shuffledQuestions = [];
let currentIndex = 0;
let isAnswered = false;

const categoryScreen = document.getElementById('categoryScreen');
const gameScreen = document.getElementById('gameScreen');
const categoryGrid = document.getElementById('categoryGrid');
const questionDisplay = document.getElementById('questionDisplay');
const optionA = document.getElementById('optionA');
const optionB = document.getElementById('optionB');
const counter = document.getElementById('counter');
const backBtn = document.getElementById('backToCategories');
const resetBtn = document.getElementById('resetCategoryBtn');

function shuffleArray(arr) {
    for (let i = arr.length - 1; i > 0; i--) {
        const j = Math.floor(Math.random() * (i + 1));
        [arr[i], arr[j]] = [arr[j], arr[i]];
    }
    return arr;
}

function startCategory(categoryName) {
    currentCategory = categoryName;
    if (categoryName === 'Все') {
        currentQuestions = [...allQuestions];
    } else {
        currentQuestions = prepareCategoryData(categoryName);
    }
    shuffledQuestions = shuffleArray([...currentQuestions]);
    currentIndex = 0;
    isAnswered = false;
    showGame();
    renderQuestion();
}

function renderQuestion() {
    if (currentIndex >= shuffledQuestions.length) {
        questionDisplay.innerHTML = `
            <div class="end-message">
                <span class="emoji-big">🎉</span>
                Вопросы в этой категории закончились!<br>
                Нажмите «Перемешать заново», чтобы начать сначала.
            </div>
        `;
        optionA.style.display = 'none';
        optionB.style.display = 'none';
        counter.textContent = `Осталось: 0`;
        return;
    }
    optionA.style.display = 'flex';
    optionB.style.display = 'flex';

    const q = shuffledQuestions[currentIndex];
    questionDisplay.textContent = q.question;
    questionDisplay.classList.remove('fade');
    void questionDisplay.offsetWidth;
    questionDisplay.classList.add('fade');

    optionA.textContent = q.optionA;
    optionB.textContent = q.optionB;
    optionA.disabled = false;
    optionB.disabled = false;
    optionA.className = 'option-btn a';
    optionB.className = 'option-btn b';
    isAnswered = false;

    const remaining = shuffledQuestions.length - currentIndex;
    counter.textContent = `Осталось: ${remaining}`;
}

function selectOption(choice) {
    if (isAnswered) return;
    if (currentIndex >= shuffledQuestions.length) return;
    isAnswered = true;

    if (choice === 'A') {
        optionA.className = 'option-btn selected-a';
        optionB.className = 'option-btn b';
    } else {
        optionB.className = 'option-btn selected-b';
        optionA.className = 'option-btn a';
    }
    optionA.disabled = true;
    optionB.disabled = true;

    setTimeout(() => {
        currentIndex++;
        renderQuestion();
    }, 700);
}

function resetCategory() {
    if (!currentCategory) return;
    const all = currentCategory === 'Все' ? [...allQuestions] : prepareCategoryData(currentCategory);
    shuffledQuestions = shuffleArray([...all]);
    currentIndex = 0;
    isAnswered = false;
    renderQuestion();
}

function showCategoryScreen() {
    categoryScreen.style.display = 'flex';
    gameScreen.style.display = 'none';
    currentCategory = '';
    currentQuestions = [];
    shuffledQuestions = [];
    currentIndex = 0;
}

function showGame() {
    categoryScreen.style.display = 'none';
    gameScreen.style.display = 'flex';
}

    };
    const sortedNames = categoryNames.slice().sort();
    sortedNames.forEach(name => {

        btn.className = 'cat-btn';

        btn.addEventListener('click', () => startCategory(name));
        categoryGrid.appendChild(btn);
    });
    const btnAll = document.createElement('button');
    btnAll.className = 'cat-btn all';
    btnAll.innerHTML = `<span class="emoji">🌟</span> Все вопросы (общая колода)`;
    btnAll.addEventListener('click', () => startCategory('Все'));
    categoryGrid.appendChild(btnAll);
}
